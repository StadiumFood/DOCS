%********************************************************getById********************************************************%
\chapter{getById}

\section*{Метод}
GET

\section*{Структура запроса}
\begin{lstlisting}
localhost:9999/type/{typeId}
\end{lstlisting}
\hfill

\section*{Параметры запроса}
\begin{table}[htbp]
    \centering
    \begin{tabularx}{\textwidth}{bsm}
    
    	\rowcolor{titleColor}
        \textbf{Параметр} & \textbf {Тип} & \textbf{Описание} \\  
        
         typeId & int  & Id типа продукта \\
    \end{tabularx}
\end{table}

\section*{Ответные данные}

\begin{table}[htbp]
    \centering
    \begin{tabularx}{\textwidth}{bsm}
    
    	\rowcolor{titleColor}
        \textbf{Параметр} & \textbf {Тип} & \textbf{Описание} \\  
        

		id & int  & Id типа продукта \\   \rowcolor{codeColor}
        name & string  & Наименование типа продукта \\   
    \end{tabularx}
\end{table}

\section*{Пример}

\subsection*{Запрос}

\begin{lstlisting}
localhost:9999/product/2
\end{lstlisting}
\hfill

\subsection*{Ответ}

\begin{lstlisting}
{
    "code": 200,
    "message": "OK",
    "data": {
        "id": 1,
        "type": "some type"
    }
}
\end{lstlisting}
\hfill
