\chapter{getById}

\section*{Метод}
GET

\section*{Структура запроса}
\begin{lstlisting}
localhost:9999/position/{positionId}
\end{lstlisting}
\hfill

\section*{Параметры запроса}
\begin{table}[htbp]
    \centering
    \begin{tabularx}{\textwidth}{bsm}
    
    	\rowcolor{titleColor}
        \textbf{Параметр} & \textbf{Тип} & \textbf{Описание} \\  
        
         positionId & int  & Id должности \\
    \end{tabularx}
\end{table}

\section*{Ответные данные}
\begin{table}[htbp]
    \centering
    \begin{tabularx}{\textwidth}{bsm}
    
    	\rowcolor{titleColor}
        \textbf{Параметр} & \textbf{Тип} & \textbf{Описание} \\  
        

		id & int  & Id должности \\   \rowcolor{codeColor}
        name & string  &  Наименование должности \\   

    \end{tabularx}
\end{table}

\section*{Пример}

\subsection*{Запрос}

\begin{lstlisting}

\end{lstlisting}
\hfill

\subsection*{Ответ}

\begin{lstlisting}
{
    "code": 200,
    "message": "OK",
    "data": {
        "id": 1,
        "name": "manager,
    }
}
\end{lstlisting}
\hfill
